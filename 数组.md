###push() , unshift() , pop() , shift() 操作原数组
```javascript
//push()
var bePush = [];
bePush.push(1);
bePush.push(2);
bePush.push(3);
console.log(bePush);
//[1,2,3]


//unshift()
var beUnshift = [];
beUnshift.unshift(1);
beUnshift.unshift(2);
beUnshift.unshift(3);
console.log(beUnshift);
//[3,2,1]


//pop()
var bePop = [1,2,3];
bePop.pop();
console.log(bePop);
//[1,2]


//shift
var beShift = [1,2,3];
beShift.shift();
console.log(beShift);
//[2,3]
```
###sort() , reverse() 操作原数组
```javascript
//reverse()
var beReverse=['a','b','c'];
beReverse.reverse();
console.log(beReverse);
//['c','b','a']


//sort(function)
var beSort=[1,3,5,2,4,6];
beSort(function(a,b){return a-b});
console.log(beSort);
//[1,2,3,4,5,6]

var beSort=[1,3,5,2,4,6];
beSort(function(a,b){return b-a});
console.log(beSort);
//[6,5,4,3,2,1]

//用sort对数组里的对象排序
var beSort=[{name:'d',age:4},{name:'b',age:2},{name:'a',age:1},{name:'c',age:3}];
beSort(function(a,b){return a.age-b.age});
console.log(beSort);
//[{name:'a',age:1},{name:'b',age:2},{name:'c',age:3},{name:'d',age:4}]
```
