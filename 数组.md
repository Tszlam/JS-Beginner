###push() , unshift() , pop() , shift() 操作原数组
```javascript
//push()从后加
var bePush = [];
bePush.push(1);
bePush.push(2);
bePush.push(3);
console.log(bePush);
//[1,2,3]


//unshift()从前加
var beUnshift = [];
beUnshift.unshift(1);
beUnshift.unshift(2);
beUnshift.unshift(3);
console.log(beUnshift);
//[3,2,1]


//pop()从后删
var bePop = [1,2,3];
bePop.pop();
console.log(bePop);
//[1,2]


//shift()从前删
var beShift = [1,2,3];
beShift.shift();
console.log(beShift);
//[2,3]
```
###sort() , reverse() 操作原数组
```javascript
//reverse()倒序
var beReverse=['a','b','c'];
beReverse.reverse();
console.log(beReverse);
//['c','b','a']


//sort(function)排序
var beSort=[1,3,5,2,4,6];
beSort(function(a,b){return a-b});
console.log(beSort);
//[1,2,3,4,5,6]

var beSort=[1,3,5,2,4,6];
beSort(function(a,b){return b-a});
console.log(beSort);
//[6,5,4,3,2,1]

//用sort对数组里的对象排序
var beSort=[{name:'d',age:4},{name:'b',age:2},{name:'a',age:1},{name:'c',age:3}];
beSort(function(a,b){return a.age-b.age});
console.log(beSort);
//[{name:'a',age:1},{name:'b',age:2},{name:'c',age:3},{name:'d',age:4}]
```
###slice() 返回新数组 , splice()操作原数组
```javascript
//slice(start[,end])截取,切的位置是元素的头
//slice(开始位置[,结束位置])
var beSlice = [0,1,2,3,4,5];
var sliceArr = beSlice.slice(2);
console.log(beSlice);
console.log(sliceArr);
//[0,1,2,3,4,5]
//[2,3,4,5]

var beSlice = [0,1,2,3,4,5];
var sliceArr = beSlice.slice(2,4);
console.log(beSlice);
console.log(sliceArr);
//[0,1,2,3,4,5]
//[2,3]


//splice(index,howmany[,item1,.....,itemX])删除与添加
//splice(删除开始位置,删除个数[,添加元素1,.....,添加元素X])
var beSplice = [0,1,2,3,4,5];
beSplice.splice(2,2,'aa','bb');//从2开始，删2个，再加上'aa'和'bb'
console.log(beSplice);
//[0,1,'aa','bb',4,5]

//用splice()剔除数组中的特定项
var Ar = [1,2,3,2,4,2,5,2,6,2,7,2,8];
for(var i=Ar.length-1;i>=0;i--){
	if(Ar[i]==2) Ar.splice(i,1);
}
console.log(Ar);
//[1,3,4,5,6,7,8]
//要从后开始，这样才不会打乱index

```
###concat() 返回新数组 , join() 返回新字符串
```javascript
//concat(arrayX,arrayX,......,arrayX)可加数组或元素
var a1 = [1,2,3];
var a2 = [6,7,8];
var anew = a1.concat(4,5,a2);
console.log(anew);
//[1,2,3,4,5,6,7,8]


//join([separator])将数组中元素连一起放进一个字符串，可选分隔符，默认为","
var a1 = ['a','b','c'];
var str1 = a1.join();
console.log(str1);
//a,b,c

var a1 = ['a','b','c'];
var str1 = a1.join("");
console.log(str1);
//abc

var a1 = ['a','b','c'];
var str1 = a1.join("--");
console.log(str1);
//a--b--c
```
###map()
```javascript
var a1 = [1,2,3];
a1.map(function(){console.log(arguments);});
//[1 , 0 , [1,2,3]]
//[2 , 1 , [1,2,3]]
//[3 , 2 , [1,2,3]]
//传入函数的参数为(value,index,array)
//可以进行遍历数组，逐一对元素进行操作
```
###indexOf()  返回第一个找的的index值，没有找到返回-1
```javascript
//indexOf(查找的值[,开始查找的位置])
var a = { foo: 'bar' };
var b = [a, 2];
console.log(b.indexOf(1))
//-1
console.log(b.indexOf({ foo: 'bar' }))
//-1
console.log(b.indexOf(a))
//0
console.log(b.indexOf(a, 1))
//-1
console.log(b.indexOf(2, 1))
//1
```
